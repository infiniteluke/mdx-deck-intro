import React from 'react'

import Circle from './Circle'

const AnimatedCircle = animated(Circle)

export default ({
  color,
  x=['50%','75%'],
  y=['50%','75%']
}) => {
  const step = useSteps(x.length-1)
  const props = useSpring({ x: x[step],y: y[step]})
  return (
    <AnimatedCircle
      x={props.x}
      y={props.y}
      r="50"
      color={color}
    />
  )
}
